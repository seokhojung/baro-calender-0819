### 1️⃣ 프로젝트 초기 설정 스크립트

```bash
#!/bin/bash
# scripts/setup.sh

set -e

echo "🚀 바로캘린더 프로젝트 설정을 시작합니다..."

# Node.js 버전 확인
node_version=$(node -v | cut -d'v' -f2 | cut -d'.' -f1)
if [ $node_version -lt 18 ]; then
  echo "❌ Node.js 18 이상이 필요합니다. 현재 버전: $(node -v)"
  exit 1
fi

echo "✅ Node.js 버전 확인 완료: $(node -v)"

# 새 Next.js 프로젝트 생성 (이미 존재하지 않는 경우)
if [ ! -f "package.json" ]; then
  echo "📦 Next.js 프로젝트를 생성합니다..."
  npx create-next-app@latest . --typescript --tailwind --eslint --app --src-dir --import-alias "@/*" --use-npm
else
  echo "📦 기존 프로젝트가 감지되었습니다."
fi

# 의존성 설치
echo "📚 의존성을 설치합니다..."

# 상태 관리 및 폼
npm install zustand react-hook-form @hookform/resolvers zod

# UI 라이브러리
npm install framer-motion lucide-react sonner

# 날짜 및 유틸리티
npm install date-fns clsx tailwind-merge class-variance-authority

# 개발 도구
npm install -D @types/node @testing-library/react @testing-library/jest-dom @testing-library/user-event
npm install -D jest jest-environment-jsdom @playwright/test
npm install -D prettier eslint-config-prettier eslint-plugin-prettier
npm install -D husky lint-staged
npm install -D @lhci/cli lighthouse
npm install -D jest-axe

# ShadCN UI 설정
echo "🎨 ShadCN UI를 설정합니다..."
npx shadcn@latest init --yes --defaults

# 기본 컴포넌트 설치
npx shadcn@latest add button input select checkbox tabs calendar card dialog sheet scroll-area badge separator form label textarea dropdown-menu popover tooltip toast alert-dialog command

# Git hooks 설정
echo "🪝 Git hooks를 설정합니다..."
npx husky init
echo "npm run lint-staged" > .husky/pre-commit
chmod +x .husky/pre-commit

# 환경 파일 생성
if [ ! -f ".env.local" ]; then
  echo "🔐 환경 파일을 생성합니다..."
  cp .env.example .env.local 2>/dev/null || echo "NEXT_PUBLIC_APP_ENV=development" > .env.local
fi

echo "✨ 프로젝트 설정이 완료되었습니다!"
echo "🎉 npm run dev를 실행하여 개발을 시작하세요."
```
